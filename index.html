    <script>

        // Wordle Game

        const wordleWord = "hello";

        let wordleAttempts = 0;


        function checkWordle() {

            const input = document.getElementById("wordle-input").value.toLowerCase();

            const feedback = document.getElementById("wordle-feedback");

            wordleAttempts++;

            

            if (input === wordleWord) {

                feedback.innerHTML = `<p>Correct! You guessed it in ${wordleAttempts} attempts.</p>`;

            } else {

                feedback.innerHTML = `<p>Incorrect, try again!</p>`;

            }

        }


        // Strands Game

        const strandsWord = "apple";

        const strandsLetters = strandsWord.split('').sort(() => Math.random() - 0.5);

        document.getElementById("strands-letters").innerHTML = strandsLetters.join(" ");


        function checkStrands() {

            const input = document.getElementById("strands-input").value.toLowerCase();

            const feedback = document.getElementById("strands-feedback");


            if (input === strandsWord) {

                feedback.innerHTML = `<p>Correct! The word is "${strandsWord}".</p>`;

            } else {

                feedback.innerHTML = `<p>Incorrect, try again!</p>`;

            }

        }


        // Connections Game

        const connectionsGroups = [

            ["apple", "banana", "cherry", "grape"],

            ["dog", "cat", "bird", "fish"]

        ];

        const connectionsItems = connectionsGroups.flat().sort(() => Math.random() - 0.5);

        const connectionsContainer = document.getElementById("connections-grid");


        connectionsItems.forEach((item, index) => {

            const button = document.createElement("button");

            button.innerText = item;

            button.onclick = () => selectConnection(button, index);

            connectionsContainer.appendChild(button);

        });


        let selectedConnections = [];

        function selectConnection(button, index) {

            button.disabled = true;

            selectedConnections.push(connectionsItems[index]);


            if (selectedConnections.length === connectionsGroups[0].length) {

                const isCorrect = connectionsGroups.some(group =>

                    group.every(item => selectedConnections.includes(item))

                );

                const feedback = document.getElementById("connections-feedback");


                if (isCorrect) {

                    feedback.innerHTML = "<p>Correct group!</p>";

                } else {

                    feedback.innerHTML = "<p>Incorrect group, try again!</p>";

                }

                selectedConnections = [];

                Array.from(connectionsContainer.children).forEach(b => (b.disabled = false));

            }

        }


        // Crossword Game

        const crosswordClues = ["A fruit (5 letters)", "A vehicle (3 letters)"];

        const crosswordAnswers = ["apple", "car"];

        const crosswordGrid = document.getElementById("crossword-grid");


        crosswordClues.forEach((clue, index) => {

            const div = document.createElement("div");

            div.innerHTML = `

                <p>${clue}</p>

                <input id="crossword-${index}" type="text" maxlength="${crosswordAnswers[index].length}" />

            `;

            crosswordGrid.appendChild(div);

        });


        function checkCrossword() {

            const feedback = document.getElementById("crossword-feedback");

            const correct = crosswordAnswers.every((answer, index) =>

                document.getElementById(`crossword-${index}`).value.toLowerCase() === answer

            );


            if (correct) {

                feedback.innerHTML = "<p>All answers are correct!</p>";

            } else {

                feedback.innerHTML = "<p>Some answers are incorrect, try again!</p>";

            }

        }
